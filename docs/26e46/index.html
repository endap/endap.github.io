Current scripts are dse for macOS and dswe for Windows. Following are the old deploy scripts.

ds

#!/bin/bash

# Buid the Hugo site into public directory
rm -rf public
hugo --minify

## Pull request when necessary
cd ../butiran
git pull
cd ../butiran-src

# Perform remote synchronization
rsync -av --delete public/ ../butiran/docs/

# Commit and push changes in the public repository
cd ../butiran
git pull
git add docs
git commit -m "rsync from butiran-src"
git push
cd ../butiran-src

#!/bin/bash

# Buid the Hugo site into public directory
rm -rf public
hugo --minify

## Pull request when necessary
cd ../butiran
git pull
cd ../butiran-src

# Perform remote synchronization
robocopy public ../butiran/docs '//MIR' '//COPY:DAT' '//R:2' '//W:2'

# Commit and push changes in the public repository
cd ../butiran
git pull
git add docs
git commit -m "robocopy from butiran-src"
git push
