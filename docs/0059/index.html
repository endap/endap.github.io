code

import random as rnd

repetition = 10
for r in range(repetition):
  x = [0]
  
  steps = 99
  for s in range(steps):
    dx = rnd.randint(0, 1) * 2 - 1
    xnew = x[-1] + dx
    x.append(xnew)
  
  print(x)
