function readFromTextarea(id, label) {
  const el = document.getElementById(id);
  let lines = el.value.split("\n");
  let val = null;
  for(l of lines) {
    let cols = l.split(" ");
    if(cols[0] == label) {
      val = cols[1];
    }
  }
  return val;
}
